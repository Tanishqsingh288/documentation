This is the NPM package which Inputs the parameters and return the Current exchanged value of the given amount.

sample usage code>>

const { convertCurrency } = require('curx');

// Example usage:
convertCurrency('USD', 'EUR', 100) // Convert 100 USD to EURO
    .then(result => console.log(`Exchanged value: €${result}`))
    .catch(error => console.log(error));
